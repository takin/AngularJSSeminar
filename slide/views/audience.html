Router

"Router berfungsi untuk mengatur view mana yang akan ditampilkan di dalam ng-view sesuai dengan konfigurasi yang ditentukan"

"Konfigurasi routing dilakukan dalam modul config dan dilakukan terhadap $routeProvider" 

// create a new module
var app = angular.module('myApp',['ngRoute']);

/* konfigurasi routing */
app.config(function($routeProvider){
  $routeProvider
  	.when('/',{
	   template:'
Hello World
'
	})
	.when('/posts', {
	   templateUrl:'views/posts.html',
	   controller: 'PostCtrl'
	})
	.when('/post/:id',{
	   templateUrl: 'views/single_post.html',
	   controller:'SinglePostCtrl',
	   resolve:{
	      return {
	      	getPost: function(BlogService, $routeParams){
	      		return BlogService.getOne($routeParams.id);
	      	}
	      }
	   }
	});

  $routeProvider.otherwise({
	redirectTo: '/'
  });
});

angular.module('myApp')
	.controller('MyController', function($scope,MyService){
		/* set title model text */
		$scope.title = "Welcome to My Blog";

		/* do some AJAX Request and assign the result */
		/* when ready to the posts model */
		MyService.getPosts().success(function(result){
			$scope.posts = result.data;
		});
	});

.directive('myDirective', function(){

Hello World

		/* rendring di layar (sebelum linking phase) */

Tips membangun aplikasi

  • Gunakan HTML untuk membangun template
  • Jangan gunakan controller untuk memanipulasi data
  • Gunakan Service/Factory untuk berkomunikasi dengan server
  • Gunakan directive untuk bagian-bagian yang reusable
  • Jangan memanipulasi data di dalam template
  • Letakkan proses yang berulang-ulang untuk setiap route di dalam service "run"
